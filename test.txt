import pandas as pd
headers = ["Name", "Customer ID", "Aadhar.No", "Phone Number", "Zipcode"]
data = [["#Batman", "01-", "59366654@4125", "9568741!124", "641?024"],
        ["#Batman", "01-", "59366654@4125", "9568741!124", "641?024"],
        ["$Hulk", "02-", "59003!!94@23146", "98**55741!004", "641?0^55"]]
df = pd.DataFrame(data, columns=headers)
print(f"Unclean Dataset:\n{df}")
df = df.drop_duplicates()
col = df.columns
for i in col:
    df[i] = df[i].str.replace(r"\W", "", regex=True)
print(f"\nClean Dataset:\n{df}")
