print("Bubble Sort:")
print("Enter 5 student ID:")
ids = []
for x in range(5):
    y = eval(input())
    ids.append(y)
print("Unsorted list is :", ids)
for i in range(len(ids)-1 , 0 , -1):
    for j in range(i):
        if ids[j]>ids[j+1]:
            temp = ids[j]
            ids[j]= ids[j+1]
            ids[j+1] = temp
print("Sorted list is :", ids)
